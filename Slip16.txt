package com.example.project1;

import android.app.Activity;
import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.*;


public class MainActivity extends Activity {

    EditText etName, etSurname, etClass, etMarks;
    RadioGroup radioGender;
    CheckBox cbReading, cbSports, cbMusic;
    Button btnSubmit;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        etName = findViewById(R.id.etName);
        etSurname = findViewById(R.id.etSurname);
        etClass = findViewById(R.id.etClass);
        etMarks = findViewById(R.id.etMarks);

        radioGender = findViewById(R.id.radioGender);
        cbReading = findViewById(R.id.cbReading);
        cbSports = findViewById(R.id.cbSports);
        cbMusic = findViewById(R.id.cbMusic);

        btnSubmit = findViewById(R.id.btnSubmit);

        btnSubmit.setOnClickListener(v -> {

            String gender = "";
            int selectedId = radioGender.getCheckedRadioButtonId();
            if (selectedId != -1) {
                RadioButton rb = findViewById(selectedId);
                gender = rb.getText().toString();
            }

            String hobbies = "";
            if (cbReading.isChecked()) hobbies += "Reading ";
            if (cbSports.isChecked()) hobbies += "Sports ";
            if (cbMusic.isChecked()) hobbies += "Music ";

            Intent i = new Intent(MainActivity.this, second.class);

            i.putExtra("name", etName.getText().toString());
            i.putExtra("surname", etSurname.getText().toString());
            i.putExtra("class", etClass.getText().toString());
            i.putExtra("gender", gender);
            i.putExtra("hobbies", hobbies);
            i.putExtra("marks", etMarks.getText().toString());

            startActivity(i);
        });
    }
}

MainActivity.java—
package com.example.project1;

import android.app.Activity;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.TextView;



public class MainActivity extends Activity {

    TextView tvResult;
    String current = "";
    double num1 = 0, num2 = 0;
    String operator = "";

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        tvResult = findViewById(R.id.tvResult);
    }

    // Number Click
    public void onClickNumber(View v) {
        Button b = (Button) v;
        current += b.getText().toString();
        tvResult.setText(current);
    }

    // Operator Click
    public void onClickOperator(View v) {
        Button b = (Button) v;
        num1 = Double.parseDouble(current);
        operator = b.getText().toString();
        current = "";
    }

    // Equal Click
    public void onClickEqual(View v) {
        num2 = Double.parseDouble(current);
        double result = 0;

        switch (operator) {
            case "+":
                result = num1 + num2;
                break;
            case "-":
                result = num1 - num2;
                break;
            case "*":
                result = num1 * num2;
                break;
            case "/":
                result = num1 / num2;
                break;
        }

        tvResult.setText(String.valueOf(result));
        current = String.valueOf(result);
    }

    // Clear
    public void onClickClear(View v) {
        current = "";
        num1 = num2 = 0;
        tvResult.setText("0");
    }
}

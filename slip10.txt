package com.example.switchtoggle;
import android.app.Activity; // Native Activity
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.Switch; // Native Switch
import android.widget.ToggleButton;
import android.widget.Toast;

public class MainActivity extends Activity { // Removed AppCompatActivity

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        final Switch sw = findViewById(R.id.mySwitch);
        final ToggleButton tg = findViewById(R.id.myToggle);
        Button btn = findViewById(R.id.btnStatus);

        btn.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                String sStatus = sw.isChecked() ? "Switch ON" : "Switch OFF";
                String tStatus = tg.isChecked() ? "Toggle ON" : "Toggle OFF";

                Toast.makeText(MainActivity.this,
                        sStatus + "\n" + tStatus, Toast.LENGTH_SHORT).show();
            }
        });
    }
}

package com.example.arrayadapter;
import android.app.Activity; // Native Activity
import android.os.Bundle;
import android.view.View;
import android.widget.AdapterView;
import android.widget.ArrayAdapter;
import android.widget.ListView;
import android.widget.Toast;

public class MainActivity extends Activity {

    // 1. Data Source
    String[] fruits = {"Apple", "Banana", "Cherry", "Date", "Elderberry", "Fig", "Grape"};

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        ListView listView = findViewById(R.id.fruitListView);

        // 2. Create the ArrayAdapter
        // We use android.R.layout.simple_list_item_1 (a built-in layout for a single text row)
        ArrayAdapter<String> adapter = new ArrayAdapter<String>(
                this,
                android.R.layout.simple_list_item_1,
                fruits
        );

        // 3. Bind the adapter to the ListView
        listView.setAdapter(adapter);

        // 4. Handle Item Clicks
        listView.setOnItemClickListener(new AdapterView.OnItemClickListener() {
            @Override
            public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
                String selectedFruit = fruits[position];
                Toast.makeText(MainActivity.this, "You selected: " + selectedFruit, Toast.LENGTH_SHORT).show();
            }
        });
    }
}

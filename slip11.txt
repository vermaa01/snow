package com.example.fontsize;


import android.app.Activity;
import android.graphics.Color;
import android.graphics.Typeface;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.TextView;

public class MainActivity extends Activity {

    float currentSize = 20;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        final TextView tv = findViewById(R.id.tvTarget);
        Button btnSize = findViewById(R.id.btnSize);
        Button btnColor = findViewById(R.id.btnColor);
        Button btnFont = findViewById(R.id.btnFont);

        // 1. Change Font Size
        btnSize.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                currentSize += 5;
                if (currentSize > 50) currentSize = 20; // Reset if too big
                tv.setTextSize(currentSize);
            }
        });

        // 2. Change Color
        btnColor.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                tv.setTextColor(Color.BLUE);
            }
        });

        // 3. Change Font Family
        btnFont.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                // Use built-in Typeface (SERIF, MONOSPACE, SANS_SERIF)
                tv.setTypeface(Typeface.MONOSPACE, Typeface.BOLD_ITALIC);
            }
        });
    }
}

main.java

package com.example.studsecactdet;


import android.app.Activity;
import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;

public class MainActivity extends Activity {
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        final EditText fn = findViewById(R.id.etFName);
        final EditText mn = findViewById(R.id.etMName);
        final EditText ln = findViewById(R.id.etLName);
        final EditText dob = findViewById(R.id.etDOB);
        final EditText add = findViewById(R.id.etAddress);
        final EditText email = findViewById(R.id.etEmail);
        Button btn = findViewById(R.id.btnSubmit);

        btn.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                Intent i = new Intent(MainActivity.this, MainActivity2.class);
                i.putExtra("f_name", fn.getText().toString());
                i.putExtra("m_name", mn.getText().toString());
                i.putExtra("l_name", ln.getText().toString());
                i.putExtra("dob", dob.getText().toString());
                i.putExtra("address", add.getText().toString());
                i.putExtra("email", email.getText().toString());
                startActivity(i);
            }
        });
    }
}

package com.example.studsecactdet;


import android.app.Activity;
import android.os.Bundle;
import android.widget.TextView;

public class MainActivity2 extends Activity {
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main2);

        TextView tv = findViewById(R.id.tvResult);

        String data = "Student Details:\n\n" +
                "Name: " + getIntent().getStringExtra("f_name") + " " +
                getIntent().getStringExtra("m_name") + " " +
                getIntent().getStringExtra("l_name") + "\n" +
                "DOB: " + getIntent().getStringExtra("dob") + "\n" +
                "Address: " + getIntent().getStringExtra("address") + "\n" +
                "Email: " + getIntent().getStringExtra("email");

        tv.setText(data);
    }
}
